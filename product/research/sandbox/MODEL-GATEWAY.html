2026-04-28 · 让 image / video / audio / embedding 走同一抽象 · 接 N 个 provider 不破架构
核心结论: 引入 3 层新抽象 · 让任意模态 × 任意 provider 通过 (provider.kind, model.name, modality) 三元组路由。
  • Modality(L4 core)· text / image / video / audio / embedding · 独立维度
  • ModelAdapter(L5 execution)· 一个 adapter = 一个 (provider × modality) 实现 · 用 Protocol 定义
  • ModelGateway(L5 execution · 单 facade)· 路由请求到对的 adapter · async/sync 统一封装
Tool 层只面对 Gateway · AgentLoop 仍 0 改动 · 加 provider 不需要改任何上层代码 · 这才是 LiteLLM / OpenRouter / Vercel AI SDK 同款"加 provider 只加一个 adapter 文件"的体感。

3.2 L5 execution · ModelAdapter Protocol(每模态一个 sub-protocol)

# execution/model_gateway/base.py
@runtime_checkable
class ModelAdapter(Protocol):
    """Base · all adapters identify themselves to the registry."""
    modality: Modality
    provider_kinds: tuple[str, ...]   # e.g. ("openai", "openai-compat")
    model_patterns: tuple[str, ...]   # e.g. ("gpt-image-",)

    async def supports(self, *, provider: LLMProvider, model: LLMModel) -> bool: ...

@runtime_checkable
class ImageAdapter(ModelAdapter, Protocol):
    modality: ClassVar[Modality] = Modality.IMAGE
    async def generate(
        self,
        request: ImageGenerationRequest,
        *,
        provider: LLMProvider,
        model: LLMModel,
    ) -> ImageGenerationResult: ...

@runtime_checkable
class VideoAdapter(ModelAdapter, Protocol):
    modality: ClassVar[Modality] = Modality.VIDEO
    async def generate(
        self,
        request: VideoGenerationRequest,
        *,
        provider: LLMProvider,
        model: LLMModel,
    ) -> VideoGenerationResult: ...

3.3 L5 execution · ModelGateway(单入口 facade)

# execution/model_gateway/__init__.py
class ModelGateway:
    """Single entry point for all generation. AgentLoop / tools call only this."""

    def __init__(self) -> None:
        self._adapters: dict[Modality, list[ModelAdapter]] = defaultdict(list)

    def register(self, adapter: ModelAdapter) -> None:
        self._adapters[adapter.modality].append(adapter)

    async def generate_image(
        self, req: ImageGenerationRequest, *, provider, model
    ) -> ImageGenerationResult:
        adapter = await self._pick(Modality.IMAGE, provider, model)
        return await adapter.generate(req, provider=provider, model=model)

    async def generate_video(...) -> VideoGenerationResult: ...
    async def generate_audio(...) -> AudioGenerationResult: ...
    async def embed_text(...) -> EmbeddingResult: ...

    async def _pick(self, modality, provider, model) -> ModelAdapter:
        for a in self._adapters[modality]:
            if await a.supports(provider=provider, model=model):
                return a
        raise NoAdapterFound(modality=modality, provider=provider.kind, model=model.name)

3.4 Adapter 实现示例 · DashScope wanx(async polling)

# execution/model_gateway/adapters/dashscope_image.py
class DashScopeImageAdapter:
    modality = Modality.IMAGE
    provider_kinds = ("aliyun",)
    model_patterns = ("wanx",)

    async def supports(self, *, provider, model) -> bool:
        return provider.kind in self.provider_kinds and any(
            p in model.name for p in self.model_patterns
        )

    async def generate(self, req, *, provider, model) -> ImageGenerationResult:
        # DashScope wanx is task-based:
        # 1. POST 创建任务 · 立即返回 task_id
        # 2. 轮询 GET tasks/{task_id} 直到 status=SUCCEEDED · 每秒 1 次 · 上限 60s
        # 3. 取出 url · 下载 png bytes
        # 这一套 polling 完全在 adapter 内 · 上层看到的还是 sync generate(req) → result
        async with httpx.AsyncClient(timeout=120) as client:
            task_id = await self._submit_task(client, req, provider, model)
            urls = await self._poll_until_done(client, task_id, provider)
            images = await self._download_all(client, urls, req)
        return ImageGenerationResult(images=images, ...)
