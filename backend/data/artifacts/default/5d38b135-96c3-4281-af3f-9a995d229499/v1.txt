def check_artifact_quality(artifact_id):
    """
    模拟检查制品质量的 Python 脚本
    """
    print(f"正在检查制品: {artifact_id}")
    
    # 模拟质量评分逻辑
    quality_score = 95
    status = "PASS" if quality_score > 90 else "FAIL"
    
    return {
        "id": artifact_id,
        "score": quality_score,
        "status": status
    }

if __name__ == "__main__":
    result = check_artifact_quality("demo-001")
    print(result)
